2. Concept

I find dream research fascinating because it delves into the unexplored aspects of the human subconscious, giving me the opportunity to glimpse our deepest thoughts and emotions, which are the parts that make us human. Understanding dreams can potentially unlock new ways to enhance our overall well-being.

“The Organic Dream: Visualizing the Unseen” is a data visualization project that transforms complex datasets of dream reports into visually compelling stories. The project aims to provide users with an intuitive and engaging way to explore data.

3. Background (Motivation)

I’ve always been interested in dream and daydream research. I have been researching reveries and the collective unconscious, especially the potential for human-machine sharing of the unconscious. To better understand archetypes that are critical for understanding the collective unconscious, I decided to create the “Organic Dreams” visualization project to explore and visualize the intricate patterns and archetypes within human dreams.

Latent Dirichlet Allocation (LDA) Topic Modelling

Above is a sample LDA on topic modelling

import pandas as pd
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.decomposition import LatentDirichletAllocation

file_path = 'data/lucid-dream.csv'  # replace with your file path
data = pd.read_csv(file_path)

data.dropna(subset=['answer_text'], inplace=True)

documents = data['answer_text'].tolist()

# Convert text data into document-term matrix
vectorizer = CountVectorizer(max_df=0.95, min_df=2, stop_words='english')
doc_term_matrix = vectorizer.fit_transform(documents)

#LDA
num_topics = 5  # Number of topics to identify
lda_model = LatentDirichletAllocation(n_components=num_topics, random_state=42)
lda_output = lda_model.fit_transform(doc_term_matrix)

feature_names = vectorizer.get_feature_names_out()
topic_keywords = []
for topic_weights in lda_model.components_:
    top_keyword_locs = (-topic_weights).argsort()[:10]  # Select top 10 keywords for each topic
    topic_keywords.append([feature_names[i] for i in top_keyword_locs])

for i, keywords in enumerate(topic_keywords):
    print(f"Topic {i + 1}: {' | '.join(keywords)}")

# Assign topics to documents
data['topic'] = lda_output.argmax(axis=1)

topic_distribution = data['topic'].value_counts(normalize=True)
print("\nTopic Distribution:")
print(topic_distribution)

topic_distribution.plot(kind='bar', figsize=(8, 6))
plt.title('Topic Distribution')
plt.xlabel('Topic')
plt.ylabel('Proportion')
plt.xticks(rotation=0)
plt.show()

import pandas as pd
import matplotlib.pyplot as plt
from textblob import TextBlob

file_path = 'data/lucid-dream.csv'
data = pd.read_csv(file_path)

data.dropna(subset=['answer_text'], inplace=True)

sentiment_scores = data['answer_text'].apply(lambda x: TextBlob(str(x)).sentiment.polarity)

# Classify sentiment as positive/negative/neutral
sentiment_classes = []
for score in sentiment_scores:
    if score > 0:
        sentiment_classes.append('Positive')
    elif score < 0:
        sentiment_classes.append('Negative')
    else:

8.2 Data Processing 

import pandas as pd
import spacy
from collections import Counter
import plotly.express as px
from plotly.subplots import make_subplots

nlp = spacy.load("en_core_web_sm")

def extract_nouns(text):
    doc = nlp(text)
    # Extract nouns
    nouns = [token.text for token in doc if token.pos_ == "NOUN"]
    return nouns

def generate_bubble_chart(filename, subplot_titles):
    df = pd.read_csv(filename)
    
    df['keyword'] = df['answer_text'].apply(lambda x: ", ".join(extract_nouns(x)))
    
    # Flatten the list of keywords
    all_keywords = [keyword for sublist in df['answer_text'].apply(extract_nouns).tolist() for keyword in sublist]
    
    keyword_counts = Counter(all_keywords) # Count frequency
    
    keywords = list(keyword_counts.keys())
    counts = list(keyword_counts.values())
   
    plot_df = pd.DataFrame({'Keyword': keywords, 'Frequency': counts})
    
    #bubble chart
    fig = px.scatter(plot_df, x=range(len(keywords)), y='Frequency', size='Frequency',
                     hover_name='Keyword', title='Bubble Chart of Extracted Keywords')
    
    return fig

def main():
    falling_dream_fig = generate_bubble_chart('data/falling-dream.csv', 'Falling Dream')
    flying_dream_fig = generate_bubble_chart('data/flying-dream.csv', 'Flying Dream')

    combined_fig = make_subplots(rows=2, cols=1, subplot_titles=('Falling Dream', 'Flying Dream'))
    combined_fig.add_trace(falling_dream_fig.data[0], row=1, col=1)
    combined_fig.add_trace(flying_dream_fig.data[0], row=2, col=1)

    combined_fig.update_layout(height=800, title_text="Dream Analysis")

    combined_fig.write_html('combined_bubble_charts.html')

if __name__ == "__main__":
    main()

9. The 5th Night: The Dream Catcher

In this visualization I want to map the relationship between dream patterns and emotions. To reach this goal, I need to:

  1. Label the dream records with their dominant emotion
  1. Extract the keyword of each line of dream record
  1. Record the frequency of each keyword
  1. Map the dominant emotion and the keywords
  1. Color code the emotions, weave everything into a dream catcher.

9.1 Data

Data Source: Dream bank, a platform where people document their dreams since 2015.

Data Size: The original dataset contains 21001 rows of dream records. I only used 10000 records for easier and faster computation.

9.2 Data Processing

In this visualization, I am using sentiment analysis to group the emotions of different dreams. Then, I’ll color-code the emotions and weave them all together into a dream catcher net. First, I ran a pre-trained emotion detection model, EmoBerta, to label each dream record with its dominant emotion.

Google Colab for this process: https://colab.research.google.com/drive/1WHw4I5utEkNx2aQ9-izi8soplKBr3Pes#scrollTo=_N5QAwV-hstL

I used senborn library to plot the distribution of emotions of dreams to get and overall idea first. And the next step is tokenization and stopword removal.

def radial_layout(graph, center, radius):
    pos = {}
    num_keywords = sum(1 for node in graph if graph.nodes[node]['type'] == 'keyword')
    angle_step = 2 * math.pi / num_keywords
    angle = 0
    
    for node in graph:
        if graph.nodes[node]['type'] == 'keyword':
            pos[node] = (center[0] + radius * math.cos(angle), center[1] + radius * math.sin(angle))
            angle += angle_step
        else:
            pos[node] = center
    return pos

center = (0, 0)
radius = 10
pos = radial_layout(G, center, radius)

# Plotly Graph Objects
edge_traces = []
for emotion, color in emotion_colors.items():
    edge_x = []
    edge_y = []
    edge_text = []
    for edge in G.edges(data=True):
        if edge[2]['color'] == color:
            x0, y0 = pos[edge[0]]
            x1, y1 = pos[edge[1]]
            edge_x.append(x0)
            edge_x.append(x1)
            edge_x.append(None)
            edge_y.append(y0)
            edge_y.append(y1)
            edge_y.append(None)
            edge_text.append(f"{edge[0]} - {edge[1]}")
    edge_trace = go.Scatter(
        x=edge_x, y=edge_y,
        line=dict(width=0.5, color=color),
        hoverinfo='text',
        hovertext=edge_text,
        mode='lines',
        legendgroup=emotion,
        showlegend=True,
        name=emotion
    )
    edge_traces.append(edge_trace)

node_x = []
node_y = []
node_text = []
node_hovertext = []
node_color = []
for node in G.nodes():
    x, y = pos[node]
    node_x.append(x)
    node_y.append(y)
    node_text.append(node)
    if G.nodes[node]['type'] == 'keyword':
        node_hovertext.append(f"{node} - {G.nodes[node]['emotion']}")
        node_color.append('lightblue')
    else:
        node_hovertext.append(node)
        node_color.append(emotion_colors[node])

node_trace = go.Scatter(
    x=node_x, y=node_y,
    mode='markers+text',
    hoverinfo='text',
    text=node_text,
    hovertext=node_hovertext,
    marker=dict(
        color=node_color,
        size=10,
        line=dict(width=2)
    ),
    textposition='top center',
    showlegend=True,
    name='showing words',
    legendgroup='showing words'
)

# Create Plotly figure
fig = go.Figure(data=[node_trace] + edge_traces,
                layout=go.Layout(
                    title='Keywords and Emotions Network',
                    showlegend=True,
                    hovermode='closest',
                    margin=dict(b=20, l=5, r=5, t=40),
                    annotations=[dict(
                        text="Interactive network graph",
                        showarrow=False,
                        xref="paper", yref="paper",
                        x=0.005, y=-0.002
                    )],
                    xaxis=dict(showgrid=False, zeroline=False, showticklabels=False),
                    yaxis=dict(showgrid=False, zeroline=False, showticklabels=False))
                )

# Save to HTML
pio.write_html(fig, file='keywords_emotions_network_modified_2.html', auto_open=True)
